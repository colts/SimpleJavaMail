025    /**
026     * Mailing tool aimed for simplicity, for sending emails of any complexity. This includes emails with plain text and/or html content,
027     * embedded images and separate attachments<br />
028     * <br />
029     * This mailing tool abstracts the javax.mail API to a higher level easy to use API. For public use, this tool only works with {@link Email}
030     * instances. <br />
031     * <br />
032     * The email message structure is built to work with all email clients and has beed tested with many different webclients as well as some
033     * mainstream client applications such as MS Outlook or Mozilla Thunderbird.<br />
034     * <br />
035     * Technically, the resulting email structure is a follows:<br />
036     * 
037     * <pre>
038     * - root
039     *      - related
040     *              - alternative
041     *                      - mail tekst
042     *                      - mail html tekst
043     *              - embedded images
044     *      - attachments
045     * </pre>
046     * 
047     * <br />
048     * Usage example:<br />
049     * 
050     * <pre>
051     * Email email = new Email();
052     * email.setFromAddress(&quot;lollypop&quot;, &quot;lolly.pop@somemail.com&quot;);
053     * email.addRecipient(&quot;Sugar Cae&quot;, &quot;sugar.cane@candystore.org&quot;, RecipientType.TO);
054     * email.setText(&quot;We should meet up!!&quot;);
055     * email.setTextHTML(&quot;&lt;b&gt;We should meet up!&lt;/b&gt;&quot;);
056     * email.setSubject(&quot;Hey&quot;);
057     * new Mailer(preconfiguredMailSession).sendMail(email);
058     * // or:
059     * new Mailer(&quot;smtp.someserver.com&quot;, 25, &quot;username&quot;, &quot;password&quot;).sendMail(email);
060     * </pre>
061     * 
062     * @author Benny Bottema
063     * @see MimeEmailMessageWrapper
064     * @see Email
065     */
066    public class Mailer {
067    
068            private static final Logger logger = Logger.getLogger(Mailer.class);

074             * @see #Mailer(Session)
075             * @see #Mailer(String, int, String, String)

080             * Email address restriction flags set either by constructor or overridden by getter by user.

084            private EmailAddressValidationCriteria emailAddressValidationCriteria;

086            /**
087             * Default constructor, stores the given mail session for later use.
088             * <p>
089             * Also defines a default email address validation criteria object, which remains true to RFC 2822, meaning allowing both domain
090             * literals and quoted identifiers (see {@link EmailAddressValidationCriteria#EmailAddressValidationCriteria(boolean, boolean)}).
091             * 
092             * @param session A preconfigured mail {@link Session} object with which a {@link Message} can be produced.
093             */
094            public Mailer(final Session session) {
095                    this.session = session;
096                    this.emailAddressValidationCriteria = new EmailAddressValidationCriteria(true, true);
097            }

099            /**
100             * Overloaded constructor which produces a new {@link Session} on the fly. Use this if you don't have a mail session configured in your
101             * web container, or Spring context etc.
102             * 
103             * @param host The address of the smtp server to be used.
104             * @param port The port of the smtp server.
105             * @param username An optional username, may be <code>null</code>.
106             * @param password An optional password, may be <code>null</code>.
107             */
108            public Mailer(final String host, final int port, final String username, final String password) {
109                    this(createMailSession(host, port, username, password));
110            }
111    
112            /**
113             * Creates a {@link Session} configured with an smtp host, port and optional username and password.
114             * 
115             * @param host The address of the smtp server to be used.
116             * @param port The port of the smtp server.
117             * @param username An optional username, may be <code>null</code>.
118             * @param password An optional password, may be <code>null</code>.
119             * @return A fully configured <code>Session</code> possibly with authentication set.
120             */
121            private static Session createMailSession(final String host, final int port, final String username, final String password) {
122                    assert host != null : "Can't send an email without host";
123                    assert password == null || username != null : "Can't have a password without username";
124    
125                    final Properties props = new Properties();
126                    props.put("mail.smtp.host", host);
127                    props.put("mail.smtp.port", port);
128    
129                    if (username != null) {
130                            props.put("mail.smtp.username", username);
131                    }
132    
133                    if (password != null) {
134                            // setup mail session to authenticate when connecting
135                            props.put("mail.smtp.auth", true);
136                            return Session.getInstance(props, new Authenticator() {
137                                    @Override
138                                    protected PasswordAuthentication getPasswordAuthentication() {
139                                            return new PasswordAuthentication(username, password);
140                                    }
141                            });
142                    } else {
143                            return Session.getInstance(props);
144                    }
145            }

147            /**
148             * Processes an {@link Email} instance into a completely configured {@link Message}, which in turn is being sent to all defined
149             * recipients.
150             * 
151             * @param email The information for the email to be sent.
152             * @throws MailException Can be thrown if an email isnt validating correctly, or some other problem occurs during connection, sending
153             *             etc.
154             * @see {@link Email}
155             * @see #validate(Email)
156             * @see #prepareMessage(Email, MimeMultipart)
157             * @see #setRecipients(Email, Message)
158             * @see #setTexts(Email, MimeMultipart)
159             * @see #setEmbeddedImages(Email, MimeMultipart)
160             * @see #setAttachments(Email, MimeMultipart)
161             */
162            public final void sendMail(final Email email)
163                            throws MailException {
164                    if (validate(email)) {
165                            try {
166                                    // create new wrapper for each mail being sent (enable sending multiple emails with one mailer)
167                                    final MimeEmailMessageWrapper messageRoot = new MimeEmailMessageWrapper();
168                                    // fill and send wrapped mime message parts
169                                    final Message message = prepareMessage(email, messageRoot.multipartRoot);
170                                    setRecipients(email, message);
171                                    setTexts(email, messageRoot.multipartAlternativeMessages);
172                                    setEmbeddedImages(email, messageRoot.multipartRelated);
173                                    setAttachments(email, messageRoot.multipartRoot);
174                                    Transport.send(message);
175                            } catch (final UnsupportedEncodingException e) {
176                                    logger.error(e.getMessage(), e);
177                                    throw new MailException(String.format(MailException.INVALID_ENCODING, e.getMessage()));
178                            } catch (final MessagingException e) {
179                                    logger.error(e.getMessage(), e);
180                                    throw new MailException(String.format(MailException.GENERIC_ERROR, e.getMessage()), e);
181                            }
182                    }
183            }
184    
185            /**
186             * Validates an {@link Email} instance. Validation fails if the subject is missing, content is missing, or no recipients are defined.
187             * 
188             * @param email The email that needs to be configured correctly.
189             * @return Always <code>true</code> (throws a {@link MailException} exception if validation fails).
190             * @throws MailException Is being thrown in any of the above causes.
191             * @see EmailValidationUtil
192             */
193            public boolean validate(final Email email)
194                            throws MailException {
195                    if (email.getText() == null && email.getTextHTML() == null) {
196                            throw new MailException(MailException.MISSING_CONTENT);
197                    } else if (email.getSubject() == null || email.getSubject().equals("")) {
198                            throw new MailException(MailException.MISSING_SUBJECT);
199                    } else if (email.getRecipients().size() == 0) {
200                            throw new MailException(MailException.MISSING_RECIPIENT);
201                    } else if (email.getFromRecipient() == null) {
202                            throw new MailException(MailException.MISSING_SENDER);
203                    } else {
204                            if (!EmailValidationUtil.isValid(email.getFromRecipient().getAddress(), emailAddressValidationCriteria)) {
205                                    throw new MailException(String.format(MailException.INVALID_SENDER, email));
206                            }
207                            for (final Recipient recipient : email.getRecipients()) {
208                                    if (!EmailValidationUtil.isValid(recipient.getAddress(), emailAddressValidationCriteria)) {
209                                            throw new MailException(String.format(MailException.INVALID_RECIPIENT, email));
210                                    }
211                            }
212                    }
213                    return true;
214            }

216            /**
217             * Creates a new {@link MimeMessage} instance and prepares it in the email structure, so that it can be filled and send.
218             * 
219             * @param email The email message from which the subject and From-address are extracted.
220             * @param multipartRoot The root of the email which holds everything (filled with some email data).
221             * @return Een geprepareerde {@link Message} instantie, klaar om gevuld en verzonden te worden.
222             * @throws MessagingException Kan gegooid worden als het message niet goed behandelt wordt.
223             * @throws UnsupportedEncodingException Zie {@link InternetAddress#InternetAddress(String, String)}.
224             */
225            private Message prepareMessage(final Email email, final MimeMultipart multipartRoot)
226                            throws MessagingException, UnsupportedEncodingException {
227                    final Message message = new MimeMessage(session);
228                    message.setSubject(email.getSubject());
229                    message.setFrom(new InternetAddress(email.getFromRecipient().getAddress(), email.getFromRecipient().getName()));
230                    message.setContent(multipartRoot);
231                    message.setSentDate(new Date());
232                    return message;
233            }
234    
235            /**
236             * Fills the {@link Message} instance with recipients from the {@link Email}.
237             * 
238             * @param email The message in which the recipients are defined.
239             * @param message The javax message that needs to be filled with recipients.
240             * @throws UnsupportedEncodingException See {@link InternetAddress#InternetAddress(String, String)}.
241             * @throws MessagingException See {@link Message#addRecipient(javax.mail.Message.RecipientType, Address)}
242             */
243            private void setRecipients(final Email email, final Message message)
244                            throws UnsupportedEncodingException, MessagingException {
245                    for (final Recipient recipient : email.getRecipients()) {
246                            final Address address = new InternetAddress(recipient.getAddress(), recipient.getName());
247                            message.addRecipient(recipient.getType(), address);
248                    }
249            }
250    
251            /**
252             * Fills the {@link Message} instance with the content bodies (text and html).
253             * 
254             * @param email The message in which the content is defined.
255             * @param multipartAlternativeMessages See {@link MimeMultipart#addBodyPart(BodyPart)}
256             * @throws MessagingException See {@link BodyPart#setText(String)}, {@link BodyPart#setContent(Object, String)} and
257             *             {@link MimeMultipart#addBodyPart(BodyPart)}.
258             */
259            private void setTexts(final Email email, final MimeMultipart multipartAlternativeMessages)
260                            throws MessagingException {
261                    if (email.getText() != null) {
262                            final MimeBodyPart messagePart = new MimeBodyPart();
263                            messagePart.setText(email.getText(), "UTF-8");
264                            multipartAlternativeMessages.addBodyPart(messagePart);
265                    }
266                    if (email.getTextHTML() != null) {
267                            final MimeBodyPart messagePartHTML = new MimeBodyPart();
268                            messagePartHTML.setContent(email.getTextHTML(), "text/html; charset=\"UTF-8\"");
269                            multipartAlternativeMessages.addBodyPart(messagePartHTML);
270                    }
271            }
272    
273            /**
274             * Fills the {@link Message} instance with the embedded images from the {@link Email}.
275             * 
276             * @param email The message in which the embedded images are defined.
277             * @param multipartRelated The branch in the email structure in which we'll stuff the embedded images.
278             * @throws MessagingException See {@link MimeMultipart#addBodyPart(BodyPart)} and
279             *             {@link #getBodyPartFromDatasource(AttachmentResource, String)}
280             */
281            private void setEmbeddedImages(final Email email, final MimeMultipart multipartRelated)
282                            throws MessagingException {
283                    for (final AttachmentResource embeddedImage : email.getEmbeddedImages()) {
284                            multipartRelated.addBodyPart(getBodyPartFromDatasource(embeddedImage, Part.INLINE));
285                    }
286            }
287    
288            /**
289             * Fills the {@link Message} instance with the attachments from the {@link Email}.
290             * 
291             * @param email The message in which the attachments are defined.
292             * @param multipartRoot The branch in the email structure in which we'll stuff the attachments.
293             * @throws MessagingException See {@link MimeMultipart#addBodyPart(BodyPart)} and
294             *             {@link #getBodyPartFromDatasource(AttachmentResource, String)}
295             */
296            private void setAttachments(final Email email, final MimeMultipart multipartRoot)
297                            throws MessagingException {
298                    for (final AttachmentResource resource : email.getAttachments()) {
299                            multipartRoot.addBodyPart(getBodyPartFromDatasource(resource, Part.ATTACHMENT));
300                    }
301            }
302    
303            /**
304             * Helper method which generates a {@link BodyPart} from an {@link AttachmentResource} (from its {@link DataSource}) and a disposition
305             * type ({@link Part#INLINE} or {@link Part#ATTACHMENT}). With this the attachment data can be converted into objects that fit in the
306             * email structure. <br />
307             * <br />
308             * For every attachment and embedded image a header needs to be set.
309             * 
310             * @param resource An object that describes the attachment and contains the actual content data.
311             * @param dispositionType The type of attachment, {@link Part#INLINE} or {@link Part#ATTACHMENT} .
312             * @return An object with the attachment data read for placement in the email structure.
313             * @throws MessagingException All BodyPart setters.
314             */
315            private BodyPart getBodyPartFromDatasource(final AttachmentResource resource, final String dispositionType)
316                            throws MessagingException {
317                    final BodyPart attachmentPart = new MimeBodyPart();
318                    final DataSource ds = resource.getDataSource();
319                    // setting headers isn't working nicely using the javax mail API, so let's do that manually
320                    attachmentPart.setDataHandler(new DataHandler(resource.getDataSource()));
321                    attachmentPart.setFileName(resource.getName());
322                    attachmentPart.setHeader("Content-Type", ds.getContentType() + "; filename=" + ds.getName() + "; name=" + ds.getName());
323                    attachmentPart.setHeader("Content-ID", String.format("<%s>", ds.getName()));
324                    attachmentPart.setDisposition(dispositionType + "; size=0");
325                    return attachmentPart;
326            }

328            /**
329             * This class conveniently wraps all necessary mimemessage parts that need to be filled with content, attachments etc. The root is
330             * ultimately sent using JavaMail.<br />
331             * <br />
332             * The constructor creates a new email message constructed from {@link MimeMultipart} as follows:
333             * 
334             * <pre>
335             * - root
336             *      - related
337             *              - alternative
338             *                      - mail tekst
339             *                      - mail html tekst
340             *              - embedded images
341             *      - attachments
342             * </pre>
343             * 
344             * @author Benny Bottema
345             */
346            private class MimeEmailMessageWrapper {
347    
348                    private final MimeMultipart multipartRoot;
349    
350                    private final MimeMultipart multipartRelated;
351    
352                    private final MimeMultipart multipartAlternativeMessages;
353    
354                    /**
355                     * Creates an email skeleton structure, so that embedded images, attachments and (html) texts are being processed properly.
356                     */
357                    MimeEmailMessageWrapper() {
358                            multipartRoot = new MimeMultipart("mixed");
359                            final MimeBodyPart contentRelated = new MimeBodyPart();
360                            multipartRelated = new MimeMultipart("related");
361                            final MimeBodyPart contentAlternativeMessages = new MimeBodyPart();
362                            multipartAlternativeMessages = new MimeMultipart("alternative");
363                            try {
364                                    // construct mail structure
365                                    multipartRoot.addBodyPart(contentRelated);
366                                    contentRelated.setContent(multipartRelated);
367                                    multipartRelated.addBodyPart(contentAlternativeMessages);
368                                    contentAlternativeMessages.setContent(multipartAlternativeMessages);
369                            } catch (final MessagingException e) {
370                                    logger.error(e.getMessage(), e);
371                                    throw new RuntimeException(e.getMessage(), e);
372                            }
373                    }
374            }

382             */

384                    this.emailAddressValidationCriteria = emailAddressValidationCriteria;
385            }
