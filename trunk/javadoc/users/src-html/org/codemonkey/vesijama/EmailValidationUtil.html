009    package org.codemonkey.vesijama;
010    
011    import java.util.regex.Pattern;
012    
013    /**
014     * Validates an email address according to <a href="http://www.ietf.org/rfc/rfc2822.txt">RFC 2822</a>, using regular expressions.
015     * <p>
016     * From the original author: <br />
017     * <blockquote> If you use this code, please keep the author information in tact and reference my site at <a
018     * href="http://www.leshazlewood.com">leshazlewood.com</a>. Thanks! </blockquote>
019     * <p>
020     * Code sanitized by Benny Bottema (kept validation 100% in tact).
021     * 
022     * @author Les Hazlewood, Benny Bottema
023     */
024    public final class EmailValidationUtil {
025    
026            /**
027             * Private constructor; this is a utility class with static methods only, not designed for extension.
028             */
029            private EmailValidationUtil() {
030                    //
031            }
032    
033            /**
034             * Validates an e-mail with default validation flags that remains <code>true</code> to RFC 2822. This means allowing both domain
035             * literals and quoted identifiers.
036             * 
037             * @param userEnteredEmailString A complete email address.
038             * @return Whether the e-mail address is compliant with RFC 2822.
039             * @see EmailAddressValidationCriteria#EmailValidationCriteria(boolean, boolean)
040             */
041            public static boolean isValid(final String email) {
042                    return isValid(email, new EmailAddressValidationCriteria(true, true));
043            }
044    
045            /**
046             * Validates an e-mail with given validation flags.
047             * 
048             * @param userEnteredEmailString A complete email address.
049             * @param emailAddressValidationCriteria A set of flags that restrict or relax RFC 2822 compliance.
050             * @return Whether the e-mail address is compliant with RFC 2822, configured using the passed in {@link EmailAddressValidationCriteria}.
051             * @see EmailAddressValidationCriteria#EmailValidationCriteria(boolean, boolean)
052             */
053            public static boolean isValid(final String email, final EmailAddressValidationCriteria emailAddressValidationCriteria) {
054                    return buildValidEmailPattern(emailAddressValidationCriteria).matcher(email).matches();
055            }
056    
057            protected static Pattern buildValidEmailPattern(EmailAddressValidationCriteria parameterObject) {
058                    // RFC 2822 2.2.2 Structured Header Field Bodies
059                    final String wsp = "[ \\t]"; // space or tab
060                    final String fwsp = wsp + "*";
061                    // RFC 2822 3.2.1 Primitive tokens
062                    final String dquote = "\\\"";
063                    // ASCII Control characters excluding white space:
064                    final String noWsCtl = "\\x01-\\x08\\x0B\\x0C\\x0E-\\x1F\\x7F";
065                    // all ASCII characters except CR and LF:
066                    final String asciiText = "[\\x01-\\x09\\x0B\\x0C\\x0E-\\x7F]";
067                    // RFC 2822 3.2.2 Quoted characters:
068                    // single backslash followed by a text char
069                    final String quotedPair = "(\\\\" + asciiText + ")";
070                    // RFC 2822 3.2.4 Atom:
071                    final String atext = "[a-zA-Z0-9\\!\\#\\$\\%\\&\\'\\*\\+\\-\\/\\=\\?\\^\\_\\`\\{\\|\\}\\~]";
072                    final String atom = fwsp + atext + "+" + fwsp;
073                    final String dotAtomText = atext + "+" + "(" + "\\." + atext + "+)*";
074                    final String dotAtom = fwsp + "(" + dotAtomText + ")" + fwsp;
075                    // RFC 2822 3.2.5 Quoted strings:
076                    // noWsCtl and the rest of ASCII except the doublequote and backslash characters:
077                    final String qtext = "[" + noWsCtl + "\\x21\\x23-\\x5B\\x5D-\\x7E]";
078                    final String qcontent = "(" + qtext + "|" + quotedPair + ")";
079                    final String quotedString = dquote + "(" + fwsp + qcontent + ")*" + fwsp + dquote;
080                    // RFC 2822 3.2.6 Miscellaneous tokens
081                    final String word = "((" + atom + ")|(" + quotedString + "))";
082                    final String phrase = word + "+"; // one or more words.
083                    // RFC 1035 tokens for domain names:
084                    final String letter = "[a-zA-Z]";
085                    final String letDig = "[a-zA-Z0-9]";
086                    final String letDigHyp = "[a-zA-Z0-9-]";
087                    final String rfcLabel = letDig + "(" + letDigHyp + "{0,61}" + letDig + ")?";
088                    final String rfc1035DomainName = rfcLabel + "(\\." + rfcLabel + ")*\\." + letter + "{2,6}";
089                    // RFC 2822 3.4 Address specification
090                    // domain text - non white space controls and the rest of ASCII chars not including [, ], or \:
091                    final String dtext = "[" + noWsCtl + "\\x21-\\x5A\\x5E-\\x7E]";
092                    final String dcontent = dtext + "|" + quotedPair;
093                    final String domainLiteral = "\\[" + "(" + fwsp + dcontent + "+)*" + fwsp + "\\]";
094                    final String rfc2822Domain = "(" + dotAtom + "|" + domainLiteral + ")";
095                    final String domain = parameterObject.isAllowDomainLiterals() ? rfc2822Domain : rfc1035DomainName;
096                    final String localPart = "((" + dotAtom + ")|(" + quotedString + "))";
097                    final String addrSpec = localPart + "@" + domain;
098                    final String angleAddr = "<" + addrSpec + ">";
099                    final String nameAddr = "(" + phrase + ")?" + fwsp + angleAddr;
100                    final String mailbox = nameAddr + "|" + addrSpec;
101                    // now compile a pattern for efficient re-use:
102                    // if we're allowing quoted identifiers or not:
103                    final String patternString = parameterObject.isAllowQuotedIdentifiers() ? mailbox : addrSpec;
104                    return Pattern.compile(patternString);
105            }
106    }
