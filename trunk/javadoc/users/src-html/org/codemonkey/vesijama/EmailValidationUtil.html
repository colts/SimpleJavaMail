009    package org.codemonkey.vesijama;
010    
011    import java.util.regex.Pattern;
012    
013    /**
014     * Validates an email address according to <a href="http://www.ietf.org/rfc/rfc2822.txt">RFC 2822</a>, using regular expressions.
015     * <p>
016     * If you use this code, please keep the author information in tact and reference my site at <a
017     * href="http://www.leshazlewood.com">leshazlewood.com</a>. Thanks! <br />
018     * <br />
019     * Code sanitized by Benny Bottema (kept validation 100% in tact).
020     * 
021     * @author Les Hazlewood, Benny Bottema
022     */
023    public final class EmailValidationUtil {
024    
025            /**
026             * Private constructor; this is a utility class with static methods only, not designed for extension.
027             */
028            private EmailValidationUtil() {
029                    //
030            }
031    
032            /**
033             * This constant states that domain literals are allowed in the email address, e.g.:
034             * <p>
035             * <tt>someone@[192.168.1.100]</tt> or <br/>
036             * <tt>john.doe@[23:33:A2:22:16:1F]</tt> or <br/>
037             * <tt>me@[my computer]</tt>
038             * </p>
039             * <p>
040             * The RFC says these are valid email addresses, but most people don't like allowing them. If you don't want to
041             * allow them, and only want to allow valid domain names (<a href="http://www.ietf.org/rfc/rfc1035.txt">RFC
042             * 1035</a>, x.y.z.com, etc), change this constant to <tt>false</tt>.
043             * <p>
044             * Its default value is <tt>true</tt> to remain RFC 2822 compliant, but you should set it depending on what you need
045             * for your application.
046             */
047            private static final boolean ALLOW_DOMAIN_LITERALS = false;
048    
049            /**
050             * This contstant states that quoted identifiers are allowed (using quotes and angle brackets around the raw
051             * address) are allowed, e.g.:
052             * <p>
053             * <tt>"John Smith" &lt;john.smith@somewhere.com&gt;</tt>
054             * <p>
055             * The RFC says this is a valid mailbox. If you don't want to allow this, because for example, you only want users
056             * to enter in a raw address (<tt>john.smith@somewhere.com</tt> - no quotes or angle brackets), then change this
057             * constant to <tt>false</tt>.
058             * <p>
059             * Its default value is <tt>true</tt> to remain RFC 2822 compliant, but you should set it depending on what you need
060             * for your application.
061             */
062            private static final boolean ALLOW_QUOTED_IDENTIFIERS = false;
063            // RFC 2822 2.2.2 Structured Header Field Bodies
064            private static final String wsp = "[ \\t]"; // space or tab
065            private static final String fwsp = wsp + "*";
066            // RFC 2822 3.2.1 Primitive tokens
067            private static final String dquote = "\\\"";
068            // ASCII Control characters excluding white space:
069            private static final String noWsCtl = "\\x01-\\x08\\x0B\\x0C\\x0E-\\x1F\\x7F";
070            // all ASCII characters except CR and LF:
071            private static final String asciiText = "[\\x01-\\x09\\x0B\\x0C\\x0E-\\x7F]";
072            // RFC 2822 3.2.2 Quoted characters:
073            // single backslash followed by a text char
074            private static final String quotedPair = "(\\\\" + asciiText + ")";
075            // RFC 2822 3.2.4 Atom:
076            private static final String atext = "[a-zA-Z0-9\\!\\#\\$\\%\\&\\'\\*\\+\\-\\/\\=\\?\\^\\_\\`\\{\\|\\}\\~]";
077            private static final String atom = fwsp + atext + "+" + fwsp;
078            private static final String dotAtomText = atext + "+" + "(" + "\\." + atext + "+)*";
079            private static final String dotAtom = fwsp + "(" + dotAtomText + ")" + fwsp;
080            // RFC 2822 3.2.5 Quoted strings:
081            // noWsCtl and the rest of ASCII except the doublequote and backslash characters:
082            private static final String qtext = "[" + noWsCtl + "\\x21\\x23-\\x5B\\x5D-\\x7E]";
083            private static final String qcontent = "(" + qtext + "|" + quotedPair + ")";
084            private static final String quotedString = dquote + "(" + fwsp + qcontent + ")*" + fwsp + dquote;
085            // RFC 2822 3.2.6 Miscellaneous tokens
086            private static final String word = "((" + atom + ")|(" + quotedString + "))";
087            private static final String phrase = word + "+"; // one or more words.
088            // RFC 1035 tokens for domain names:
089            private static final String letter = "[a-zA-Z]";
090            private static final String letDig = "[a-zA-Z0-9]";
091            private static final String letDigHyp = "[a-zA-Z0-9-]";
092            private static final String rfcLabel = letDig + "(" + letDigHyp + "{0,61}" + letDig + ")?";
093            private static final String rfc1035DomainName = rfcLabel + "(\\." + rfcLabel + ")*\\." + letter + "{2,6}";
094            // RFC 2822 3.4 Address specification
095            // domain text - non white space controls and the rest of ASCII chars not including [, ], or \:
096            private static final String dtext = "[" + noWsCtl + "\\x21-\\x5A\\x5E-\\x7E]";
097            private static final String dcontent = dtext + "|" + quotedPair;
098            private static final String domainLiteral = "\\[" + "(" + fwsp + dcontent + "+)*" + fwsp + "\\]";
099            private static final String rfc2822Domain = "(" + dotAtom + "|" + domainLiteral + ")";
100            private static final String domain = ALLOW_DOMAIN_LITERALS ? rfc2822Domain : rfc1035DomainName;
101            private static final String localPart = "((" + dotAtom + ")|(" + quotedString + "))";
102            private static final String addrSpec = localPart + "@" + domain;
103            private static final String angleAddr = "<" + addrSpec + ">";
104            private static final String nameAddr = "(" + phrase + ")?" + fwsp + angleAddr;
105            private static final String mailbox = nameAddr + "|" + addrSpec;
106            // now compile a pattern for efficient re-use:
107            // if we're allowing quoted identifiers or not:
108            private static final String patternString = ALLOW_QUOTED_IDENTIFIERS ? mailbox : addrSpec;
109            public static final Pattern VALID_PATTERN = Pattern.compile(patternString);
110    
111            public static boolean isValid(final String userEnteredEmailString) {
112                    return VALID_PATTERN.matcher(userEnteredEmailString).matches();
113            }
114    }
