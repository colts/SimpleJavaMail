001    package org.codemonkey.vesijama;
002    
003    import javax.mail.MessagingException;
004    
005    /**
006     * This exception is used to communicate errors during the sending of email.
007     * 
008     * @author Benny Bottema
009     */
010    @SuppressWarnings("serial")
011    public final class MailException extends RuntimeException {
012    
013            protected static final String GENERIC_ERROR = "Generic error: %s";
014            protected static final String INVALID_ENCODING = "Encoding not accepted: %s";
015            protected static final String INVALID_RECIPIENT = "Invalid TO address: %s";
016            protected static final String INVALID_SENDER = "Invalid FROM address: %s";
017            protected static final String MISSING_SENDER = "Email is not valid: missing sender";
018            protected static final String MISSING_RECIPIENT = "Email is not valid: missing recipients";
019            protected static final String MISSING_SUBJECT = "Email is not valid: missing subject";
020            protected static final String MISSING_CONTENT = "Email is not valid: missing content body";
021    
022            protected MailException(final String message) {
023                    super(message);
024            }
025    
026            protected MailException(final String message, final MessagingException cause) {
027                    super(message, cause);
028            }
029    }
