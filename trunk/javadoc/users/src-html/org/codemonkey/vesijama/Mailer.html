025    /**
026     * Mailing tool aimed for simplicity, for sending emails of any complexity. This includes emails with plain text and/or
027     * html content, embedded images and separate attachments<br />
028     * <br />
029     * This mailing tool abstracts the javax.mail API to a higher level easy to use API. For public use, this tool only
030     * works with {@link Email} instances. <br />
031     * <br />
032     * The email message structure is built to work with all email clients and has beed tested with many different
033     * webclients as well as some mainstream client applications such as MS Outlook or Mozilla Thunderbird.<br />
034     * <br />
035     * Technically, the resulting email structure is a follows:<br />
036     * 
037     * <pre>
038     * - root
039     *      - related
040     *              - alternative
041     *                      - mail tekst
042     *                      - mail html tekst
043     *              - embedded images
044     *      - attachments
045     * </pre>
046     * 
047     * <br />
048     * Usage example:<br />
049     * 
050     * <pre>
051     * Email email = new Email();
052     * email.setFromAddress(&quot;lollypop&quot;, &quot;lolly.pop@somemail.com&quot;);
053     * email.addRecipient(&quot;Sugar Cae&quot;, &quot;sugar.cane@candystore.org&quot;, RecipientType.TO);
054     * email.setText(&quot;We should meet up!!&quot;);
055     * email.setTextHTML(&quot;&lt;b&gt;We should meet up!&lt;/b&gt;&quot;);
056     * email.setSubject(&quot;Hey&quot;);
057     * new Mailer(preconfiguredMailSession).sendMail(email);
058     * // or:
059     * new Mailer(&quot;smtp.someserver.com&quot;, 25, &quot;username&quot;, &quot;password&quot;).sendMail(email);
060     * </pre>
061     * 
062     * @author Benny Bottema
063     * @see Email
064     */
065    public class Mailer {
066    
067            private static final Logger logger = Logger.getLogger(Mailer.class);
068    
069            private final MimeMultipart multipartRoot;
070    
071            private final MimeMultipart multipartRelated;
072    
073            private final MimeMultipart multipartAlternativeMessages;
074    
075            /**
076             * Used to actually send the email. This session can come from being passed in the default constructor, or made by
077             * this mailer directly, when no <code>Session</code> was provided.
078             * 
079             * @see #Mailer(Session)
080             * @see #Mailer(String, int, String, String)
081             */
082            private final Session session;
083    
084            /**
085             * Default constructor. Creates an email skeleton structure, so that embedded images, attachments en (html) textx
086             * are being processed properly.
087             * 
088             * @param session A preconfigured mail {@link Session} object with which a {@link Message} can be produced.
089             */
090            public Mailer(final Session session) {
091                    this.session = session;
092                    multipartRoot = new MimeMultipart("mixed");
093                    final MimeBodyPart contentRelated = new MimeBodyPart();
094                    multipartRelated = new MimeMultipart("related");
095                    final MimeBodyPart contentAlternativeMessages = new MimeBodyPart();
096                    multipartAlternativeMessages = new MimeMultipart("alternative");
097                    try {
098                            // construct mail structure
099                            multipartRoot.addBodyPart(contentRelated);
100                            contentRelated.setContent(multipartRelated);
101                            multipartRelated.addBodyPart(contentAlternativeMessages);
102                            contentAlternativeMessages.setContent(multipartAlternativeMessages);
103                    } catch (final MessagingException e) {
104                            logger.error(e.getMessage(), e);
105                            throw new RuntimeException(e.getMessage(), e);
106                    }
107            }
108    
109            /**
110             * Overloaded constructor which produces a new {@link Session} on the fly. Use this if you don't have a mail session
111             * configured in your web container, or Spring context etc.
112             * 
113             * @param host The address of the smtp server to be used.
114             * @param port The port of the smtp server.
115             * @param username An optional username, may be <code>null</code>.
116             * @param password An optional password, may be <code>null</code>.
117             */
118            public Mailer(final String host, final int port, final String username, final String password) {
119                    this(createMailSession(host, port, username, password));
120            }
121    
122            /**
123             * Creates a {@link Session} configured with an smtp host, port and optional username and password.
124             * 
125             * @param host The address of the smtp server to be used.
126             * @param port The port of the smtp server.
127             * @param username An optional username, may be <code>null</code>.
128             * @param password An optional password, may be <code>null</code>.
129             * @return A fully configured <code>Session</code> possibly with authentication set.
130             */
131            private static Session createMailSession(final String host, final int port, final String username,
132                            final String password) {
133                    assert host != null : "Can't send an email without host";
134                    assert password == null || username != null : "Can't have a password without username";
135    
136                    final Properties props = new Properties();
137                    props.put("mail.smtp.host", host);
138                    props.put("mail.smtp.port", port);
139    
140                    if (username != null) {
141                            props.put("mail.smtp.username", username);
142                    }
143    
144                    if (password != null) {
145                            // setup mail session to authenticate when connecting
146                            props.put("mail.smtp.auth", true);
147                            return Session.getInstance(props, new Authenticator() {
148                                    @Override
149                                    protected PasswordAuthentication getPasswordAuthentication() {
150                                            return new PasswordAuthentication(username, password);
151                                    }
152                            });
153                    } else {
154                            return Session.getInstance(props);
155                    }
156            }
157    
158            /**
159             * Processed an {@link Email} instance into a completely configured {@link Message}, which in turn is being sent to
160             * all defined recipients.
161             * 
162             * @param email The information for the email to be sent.
163             * @throws MailException Can be thrown if an email isnt validating correctly, or some other problem occurs during
164             *             connection, sending etc.
165             * @see {@link Email}
166             * @see #validate(Email)
167             * @see #prepareMessage(Email, MimeMultipart)
168             * @see #setRecipients(Email, Message)
169             * @see #setTexts(Email, MimeMultipart)
170             * @see #setEmbeddedImages(Email, MimeMultipart)
171             * @see #setAttachments(Email, MimeMultipart)
172             */
173            public final void sendMail(final Email email)
174                            throws MailException {
175                    if (validate(email)) {
176                            try {
177                                    final Message message = prepareMessage(email, multipartRoot);
178                                    setRecipients(email, message);
179                                    setTexts(email, multipartAlternativeMessages);
180                                    setEmbeddedImages(email, multipartRelated);
181                                    setAttachments(email, multipartRoot);
182                                    Transport.send(message);
183                            } catch (final UnsupportedEncodingException e) {
184                                    logger.error(e.getMessage(), e);
185                                    throw new MailException(String.format(MailException.INVALID_ENCODING, e.getMessage()));
186                            } catch (final MessagingException e) {
187                                    logger.error(e.getMessage(), e);
188                                    throw new MailException(String.format(MailException.GENERIC_ERROR, e.getMessage()), e);
189                            }
190                    }
191            }
192    
193            /**
194             * Validates an {@link Email} instance. Validation fails if the subject is missing, content is missing, or no
195             * recipients are defined.
196             * 
197             * @param email The email that needs to be configured correctly.
198             * @return Always <code>true</code> (throws a {@link MailException} exception if validation fails).
199             * @throws MailException Is being thrown in any of the above causes.
200             * @see {@link EmailValidationUtil}
201             */
202            public boolean validate(final Email email)
203                            throws MailException {
204                    if (email.getText() == null && email.getTextHTML() == null) {
205                            throw new MailException(MailException.MISSING_CONTENT);
206                    } else if (email.getSubject() == null || email.getSubject().equals("")) {
207                            throw new MailException(MailException.MISSING_SUBJECT);
208                    } else if (email.getRecipients().size() == 0) {
209                            throw new MailException(MailException.MISSING_RECIPIENT);
210                    } else if (email.getFromRecipient() == null) {
211                            throw new MailException(MailException.MISSING_SENDER);
212                    } else {
213                            if (!EmailValidationUtil.isValid(email.getFromRecipient().getAddress())) {
214                                    throw new MailException(String.format(MailException.INVALID_SENDER, email));
215                            }
216                            for (final Recipient recipient : email.getRecipients()) {
217                                    if (!EmailValidationUtil.isValid(recipient.getAddress())) {
218                                            throw new MailException(String.format(MailException.INVALID_RECIPIENT, email));
219                                    }
220                            }
221                    }
222                    return true;
223            }

225            /**
226             * Creates a new {@link MimeMessage} instance and prepares it in the email structure, so that it can be filled and
227             * send.
228             * 
229             * @param email The email message from which the subject and From-address are extracted.
230             * @param multipartRoot The root of the email which holds everything (filled with some email data).
231             * @return Een geprepareerde {@link Message} instantie, klaar om gevuld en verzonden te worden.
232             * @throws MessagingException Kan gegooid worden als het message niet goed behandelt wordt.
233             * @throws UnsupportedEncodingException Zie {@link InternetAddress#InternetAddress(String, String)}.
234             */
235            private Message prepareMessage(final Email email, final MimeMultipart multipartRoot)
236                            throws MessagingException, UnsupportedEncodingException {
237                    final Message message = new MimeMessage(session);
238                    message.setSubject(email.getSubject());
239                    message.setFrom(new InternetAddress(email.getFromRecipient().getAddress(), email.getFromRecipient().getName()));
240                    message.setContent(multipartRoot);
241                    message.setSentDate(new Date());
242                    return message;
243            }
244    
245            /**
246             * Fills the {@link Message} instance with recipients from the {@link Email}.
247             * 
248             * @param email The message in which the recipients are defined.
249             * @param message The javax message that needs to be filled with recipients.
250             * @throws UnsupportedEncodingException See {@link InternetAddress#InternetAddress(String, String)}.
251             * @throws MessagingException See {@link Message#addRecipient(javax.mail.Message.RecipientType, Address)}
252             */
253            private void setRecipients(final Email email, final Message message)
254                            throws UnsupportedEncodingException, MessagingException {
255                    for (final Recipient recipient : email.getRecipients()) {
256                            final Address address = new InternetAddress(recipient.getAddress(), recipient.getName());
257                            message.addRecipient(recipient.getType(), address);
258                    }
259            }
260    
261            /**
262             * Fills the {@link Message} instance with the content bodies (text and html).
263             * 
264             * @param email The message in which the content is defined.
265             * @param multipartAlternativeMessages See {@link MimeMultipart#addBodyPart(BodyPart)}
266             * @throws MessagingException See {@link BodyPart#setText(String)}, {@link BodyPart#setContent(Object, String)} and
267             *             {@link MimeMultipart#addBodyPart(BodyPart)}.
268             */
269            private void setTexts(final Email email, final MimeMultipart multipartAlternativeMessages)
270                            throws MessagingException {
271                    final BodyPart messagePart = new MimeBodyPart();
272                    final BodyPart messagePartHTML = new MimeBodyPart();
273                    messagePart.setText(email.getText());
274                    messagePartHTML.setContent(email.getTextHTML(), "text/html");
275                    multipartAlternativeMessages.addBodyPart(messagePart);
276                    multipartAlternativeMessages.addBodyPart(messagePartHTML);
277            }
278    
279            /**
280             * Fills the {@link Message} instance with the embedded images from the {@link Email}.
281             * 
282             * @param email The message in which the embedded images are defined.
283             * @param multipartRelated The branch in the email structure in which we'll stuff the embedded images.
284             * @throws MessagingException See {@link MimeMultipart#addBodyPart(BodyPart)} and
285             *             {@link #getBodyPartFromDatasource(AttachmentResource, String)}
286             */
287            private void setEmbeddedImages(final Email email, final MimeMultipart multipartRelated)
288                            throws MessagingException {
289                    for (final AttachmentResource embeddedImage : email.getEmbeddedImages()) {
290                            multipartRelated.addBodyPart(getBodyPartFromDatasource(embeddedImage, Part.INLINE));
291                    }
292            }
293    
294            /**
295             * Fills the {@link Message} instance with the attachments from the {@link Email}.
296             * 
297             * @param email The message in which the attachments are defined.
298             * @param multipartRoot The branch in the email structure in which we'll stuff the attachments.
299             * @throws MessagingException See {@link MimeMultipart#addBodyPart(BodyPart)} and
300             *             {@link #getBodyPartFromDatasource(AttachmentResource, String)}
301             */
302            private void setAttachments(final Email email, final MimeMultipart multipartRoot)
303                            throws MessagingException {
304                    for (final AttachmentResource resource : email.getAttachments()) {
305                            multipartRoot.addBodyPart(getBodyPartFromDatasource(resource, Part.ATTACHMENT));
306                    }
307            }
308    
309            /**
310             * Helper method which generates a {@link BodyPart} from an {@link AttachmentResource} (from its {@link DataSource})
311             * and a disposition type ({@link Part#INLINE} or {@link Part#ATTACHMENT}). With this the attachment data can be
312             * converted into objects that fit in the email structure. <br />
313             * <br />
314             * For every attachment and embedded image a header needs to be set.
315             * 
316             * @param resource An object that describes the attachment and contains the actual content data.
317             * @param dispositionType The type of attachment, {@link Part#INLINE} or {@link Part#ATTACHMENT} .
318             * @return An object with the attachment data read for placement in the email structure.
319             * @throws MessagingException All BodyPart setters.
320             */
321            private BodyPart getBodyPartFromDatasource(final AttachmentResource resource, final String dispositionType)
322                            throws MessagingException {
323                    final BodyPart attachmentPart = new MimeBodyPart();
324                    final DataSource ds = resource.getDataSource();
325                    // setting headers isn't working nicely using the javax mail API, so let's do that manually
326                    attachmentPart.setDataHandler(new DataHandler(resource.getDataSource()));
327                    attachmentPart.setFileName(resource.getName());
328                    attachmentPart.setHeader("Content-Type", ds.getContentType() + "; filename=" + ds.getName() + "; name="
329                                    + ds.getName());
330                    attachmentPart.setHeader("Content-ID", ds.getName());
331                    attachmentPart.setDisposition(dispositionType + "; size=0");
332                    return attachmentPart;
333            }
