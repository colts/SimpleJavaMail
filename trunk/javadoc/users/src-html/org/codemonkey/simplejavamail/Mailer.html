025    /**
026     * Mailing tool aimed for simplicity, for sending e-mails of any complexity. This includes e-mails with plain text and/or html content,
027     * embedded images and separate attachments, SMTP, SMTPS / SSL and SMTP + SSL<br />
028     * <br />
029     * This mailing tool abstracts the javax.mail API to a higher level easy to use API. For public use, this tool only works with {@link Email}
030     * instances. <br />
031     * <br />
032     * The e-mail message structure is built to work with all e-mail clients and has been tested with many different webclients as well as some
033     * mainstream client applications such as MS Outlook or Mozilla Thunderbird.<br />
034     * <br />
035     * Technically, the resulting email structure is a follows:<br />
036     * 
037     * <pre>
038     * - root
039     *      - related
040     *              - alternative
041     *                      - mail text
042     *                      - mail html text
043     *              - embedded images
044     *      - attachments
045     * </pre>
046     * 
047     * <br />
048     * Usage example:<br />
049     * 
050     * <pre>
051     * Email email = new Email();
052     * email.setFromAddress(&quot;lollypop&quot;, &quot;lolly.pop@somemail.com&quot;);
053     * email.addRecipient(&quot;Sugar Cae&quot;, &quot;sugar.cane@candystore.org&quot;, RecipientType.TO);
054     * email.setText(&quot;We should meet up!!&quot;);
055     * email.setTextHTML(&quot;&lt;b&gt;We should meet up!&lt;/b&gt;&quot;);
056     * email.setSubject(&quot;Hey&quot;);
057     * new Mailer(preconfiguredMailSession).sendMail(email);
058     * // or:
059     * new Mailer(&quot;smtp.someserver.com&quot;, 25, &quot;username&quot;, &quot;password&quot;).sendMail(email);
060     * </pre>
061     * 
062     * @author Benny Bottema
063     * @see MimeEmailMessageWrapper
064     * @see Email
065     */
066    public class Mailer {
067    
068            private static final Logger logger = Logger.getLogger(Mailer.class);

092            /**
093             * Default constructor, stores the given mail session for later use. Assumes that *all* properties used to make a connection are
094             * configured (host, port, authentication and transport protocol settings).
095             * <p>
096             * Also defines a default email address validation criteria object, which remains true to RFC 2822, meaning allowing both domain
097             * literals and quoted identifiers (see {@link EmailAddressValidationCriteria#EmailAddressValidationCriteria(boolean, boolean)}).
098             * 
099             * @param session A preconfigured mail {@link Session} object with which a {@link Message} can be produced.
100             */
101            public Mailer(final Session session) {
102                    this.session = session;
103                    this.emailAddressValidationCriteria = new EmailAddressValidationCriteria(true, true);
104            }
105    
106            /**
107             * Overloaded constructor which produces a new {@link Session} on the fly. Use this if you don't have a mail session configured in your
108             * web container, or Spring context etc.
109             * <p>
110             * Also defines a default email address validation criteria object, which remains true to RFC 2822, meaning allowing both domain
111             * literals and quoted identifiers (see {@link EmailAddressValidationCriteria#EmailAddressValidationCriteria(boolean, boolean)}).
112             * 
113             * @param host The address URL of the SMTP server to be used.
114             * @param port The port of the SMTP server.
115             * @param username An optional username, may be <code>null</code>.
116             * @param password An optional password, may be <code>null</code>, but only if username is <code>null</code> as well.
117             * @param transportStrategy The transport protocol configuration type for handling SSL or TLS (or vanilla SMTP)
118             */
119            public Mailer(final String host, final int port, final String username, final String password, final TransportStrategy transportStrategy) {
120                    // we're doing these validations manually instead of using Apache Commons to avoid another dependency
121                    if (host == null || "".equals(host.trim())) {
122                            throw new RuntimeException("Can't send an email without host");
123                    } else if ((password != null && !"".equals(password.trim())) && (username == null || "".equals(username.trim()))) {
124                            throw new RuntimeException("Can't have a password without username");
125                    }
126                    this.transportStrategy = transportStrategy;
127                    this.session = createMailSession(host, port, username, password);
128                    this.emailAddressValidationCriteria = new EmailAddressValidationCriteria(true, true);
129            }
130    
131            /**
132             * Actually instantiates and configures the {@link Session} instance. Delegates resolving transport protocol specific properties to the
133             * {@link #transportStrategy} in two ways:
134             * <ol>
135             * <li>request an initial property list which the strategy may pre-populate</li>
136             * <li>by requesting the property names according to the respective transport protocol it handles (for the host property name it would
137             * be <em>"mail.smtp.host"</em> for SMTP and <em>"mail.smtps.host"</em> for SMTPS)</li>
138             * </ol>
139             * 
140             * @param host The address URL of the SMTP server to be used.
141             * @param port The port of the SMTP server.
142             * @param username An optional username, may be <code>null</code>.
143             * @param password An optional password, may be <code>null</code>.
144             * @return A fully configured <code>Session</code> instance complete with transport protocol settings.
145             * @see TransportStrategy#generateProperties()
146             * @see TransportStrategy#propertyNameHost()
147             * @see TransportStrategy#propertyNamePort()
148             * @see TransportStrategy#propertyNameUsername()
149             * @see TransportStrategy#propertyNameAuthenticate()
150             */
151            public Session createMailSession(final String host, final int port, final String username, final String password) {
152                    Properties props = transportStrategy.generateProperties();
153                    props.put(transportStrategy.propertyNameHost(), host);
154                    props.put(transportStrategy.propertyNamePort(), String.valueOf(port));
155    
156                    if (username != null) {
157                            props.put(transportStrategy.propertyNameUsername(), username);
158                    }
159    
160                    if (password != null) {
161                            props.put(transportStrategy.propertyNameAuthenticate(), "true");
162                            return Session.getInstance(props, new Authenticator() {
163                                    @Override
164                                    protected PasswordAuthentication getPasswordAuthentication() {
165                                            return new PasswordAuthentication(username, password);
166                                    }
167                            });
168                    } else {
169                            return Session.getInstance(props);
170                    }
171            }

189            /**
190             * Processes an {@link Email} instance into a completely configured {@link Message}.
191             * <p>
192             * Sends the Sun JavaMail {@link Message} object using {@link Session#getTransport()}. It will call {@link Transport#connect()} assuming
193             * all connection details have been configured in the provided {@link Session} instance.
194             * <p>
195             * Performs a call to {@link Message#saveChanges()} as the Sun JavaMail API indicates it is needed to configure the message headers and
196             * providing a message id.
197             * 
198             * @param email The information for the email to be sent.
199             * @throws MailException Can be thrown if an email isn't validating correctly, or some other problem occurs during connection, sending
200             *             etc.
201             * @see #validate(Email)
202             * @see #prepareMessage(Email, MimeMultipart)
203             * @see #setRecipients(Email, Message)
204             * @see #setTexts(Email, MimeMultipart)
205             * @see #setEmbeddedImages(Email, MimeMultipart)
206             * @see #setAttachments(Email, MimeMultipart)
207             */
208            public final void sendMail(final Email email)
209                            throws MailException {
210                    if (validate(email)) {
211                            try {
212                                    // create new wrapper for each mail being sent (enable sending multiple emails with one mailer)
213                                    final MimeEmailMessageWrapper messageRoot = new MimeEmailMessageWrapper();
214                                    // fill and send wrapped mime message parts
215                                    final Message message = prepareMessage(email, messageRoot.multipartRoot);
216                                    setRecipients(email, message);
217                                    setTexts(email, messageRoot.multipartAlternativeMessages);
218                                    setEmbeddedImages(email, messageRoot.multipartRelated);
219                                    setAttachments(email, messageRoot.multipartRoot);
220                                    logSession(session, transportStrategy);
221                                    message.saveChanges(); // some headers and id's will be set for this specific message
222                                    Transport transport = session.getTransport();
223                                    transport.connect();
224                                    transport.sendMessage(message, message.getAllRecipients());
225                                    transport.close();
226                            } catch (final UnsupportedEncodingException e) {
227                                    logger.error(e.getMessage(), e);
228                                    throw new MailException(String.format(MailException.INVALID_ENCODING, e.getMessage()));
229                            } catch (final MessagingException e) {
230                                    logger.error(e.getMessage(), e);
231                                    throw new MailException(String.format(MailException.GENERIC_ERROR, e.getMessage()), e);
232                            }
233                    }
234            }

236            /**
237             * Simply logs host details, credentials used and whether authentication will take place and finally the transport protocol used.
238             */
239            private void logSession(Session session, TransportStrategy transportStrategy) {
240                    final String logmsg = "starting mail session (host: %s, port: %s, username: %s, authenticate: %s, transport: %s)";
241                    Properties properties = session.getProperties();
242                    logger.debug(String.format(logmsg, properties.get(transportStrategy.propertyNameHost()),
243                                    properties.get(transportStrategy.propertyNamePort()), properties.get(transportStrategy.propertyNameUsername()),
244                                    properties.get(transportStrategy.propertyNameAuthenticate()), transportStrategy));
245            }

247            /**
248             * Validates an {@link Email} instance. Validation fails if the subject is missing, content is missing, or no recipients are defined.
249             * 
250             * @param email The email that needs to be configured correctly.
251             * @return Always <code>true</code> (throws a {@link MailException} exception if validation fails).
252             * @throws MailException Is being thrown in any of the above causes.
253             * @see EmailValidationUtil
254             */
255            public boolean validate(final Email email)
256                            throws MailException {
257                    if (email.getText() == null && email.getTextHTML() == null) {
258                            throw new MailException(MailException.MISSING_CONTENT);
259                    } else if (email.getSubject() == null || email.getSubject().equals("")) {
260                            throw new MailException(MailException.MISSING_SUBJECT);
261                    } else if (email.getRecipients().size() == 0) {
262                            throw new MailException(MailException.MISSING_RECIPIENT);
263                    } else if (email.getFromRecipient() == null) {
264                            throw new MailException(MailException.MISSING_SENDER);
265                    } else {
266                            if (!EmailValidationUtil.isValid(email.getFromRecipient().getAddress(), emailAddressValidationCriteria)) {
267                                    throw new MailException(String.format(MailException.INVALID_SENDER, email));
268                            }
269                            for (final Recipient recipient : email.getRecipients()) {
270                                    if (!EmailValidationUtil.isValid(recipient.getAddress(), emailAddressValidationCriteria)) {
271                                            throw new MailException(String.format(MailException.INVALID_RECIPIENT, email));
272                                    }
273                            }
274                    }
275                    return true;
276            }

278            /**
279             * Creates a new {@link MimeMessage} instance and prepares it in the email structure, so that it can be filled and send.
280             * 
281             * @param email The email message from which the subject and From-address are extracted.
282             * @param multipartRoot The root of the email which holds everything (filled with some email data).
283             * @return Een geprepareerde {@link Message} instantie, klaar om gevuld en verzonden te worden.
284             * @throws MessagingException Kan gegooid worden als het message niet goed behandelt wordt.
285             * @throws UnsupportedEncodingException Zie {@link InternetAddress#InternetAddress(String, String)}.
286             */
287            private Message prepareMessage(final Email email, final MimeMultipart multipartRoot)
288                            throws MessagingException, UnsupportedEncodingException {
289                    final Message message = new MimeMessage(session);
290                    message.setSubject(email.getSubject());
291                    message.setFrom(new InternetAddress(email.getFromRecipient().getAddress(), email.getFromRecipient().getName()));
292                    message.setContent(multipartRoot);
293                    message.setSentDate(new Date());
294                    return message;
295            }

297            /**
298             * Fills the {@link Message} instance with recipients from the {@link Email}.
299             * 
300             * @param email The message in which the recipients are defined.
301             * @param message The javax message that needs to be filled with recipients.
302             * @throws UnsupportedEncodingException See {@link InternetAddress#InternetAddress(String, String)}.
303             * @throws MessagingException See {@link Message#addRecipient(javax.mail.Message.RecipientType, Address)}
304             */
305            private void setRecipients(final Email email, final Message message)
306                            throws UnsupportedEncodingException, MessagingException {
307                    for (final Recipient recipient : email.getRecipients()) {
308                            final Address address = new InternetAddress(recipient.getAddress(), recipient.getName());
309                            message.addRecipient(recipient.getType(), address);
310                    }
311            }

313            /**
314             * Fills the {@link Message} instance with the content bodies (text and html).
315             * 
316             * @param email The message in which the content is defined.
317             * @param multipartAlternativeMessages See {@link MimeMultipart#addBodyPart(BodyPart)}
318             * @throws MessagingException See {@link BodyPart#setText(String)}, {@link BodyPart#setContent(Object, String)} and
319             *             {@link MimeMultipart#addBodyPart(BodyPart)}.
320             */
321            private void setTexts(final Email email, final MimeMultipart multipartAlternativeMessages)
322                            throws MessagingException {
323                    if (email.getText() != null) {
324                            final MimeBodyPart messagePart = new MimeBodyPart();
325                            messagePart.setText(email.getText(), "UTF-8");
326                            multipartAlternativeMessages.addBodyPart(messagePart);
327                    }
328                    if (email.getTextHTML() != null) {
329                            final MimeBodyPart messagePartHTML = new MimeBodyPart();
330                            messagePartHTML.setContent(email.getTextHTML(), "text/html; charset=\"UTF-8\"");
331                            multipartAlternativeMessages.addBodyPart(messagePartHTML);
332                    }
333            }

335            /**
336             * Fills the {@link Message} instance with the embedded images from the {@link Email}.
337             * 
338             * @param email The message in which the embedded images are defined.
339             * @param multipartRelated The branch in the email structure in which we'll stuff the embedded images.
340             * @throws MessagingException See {@link MimeMultipart#addBodyPart(BodyPart)} and
341             *             {@link #getBodyPartFromDatasource(AttachmentResource, String)}
342             */
343            private void setEmbeddedImages(final Email email, final MimeMultipart multipartRelated)
344                            throws MessagingException {
345                    for (final AttachmentResource embeddedImage : email.getEmbeddedImages()) {
346                            multipartRelated.addBodyPart(getBodyPartFromDatasource(embeddedImage, Part.INLINE));
347                    }
348            }

350            /**
351             * Fills the {@link Message} instance with the attachments from the {@link Email}.
352             * 
353             * @param email The message in which the attachments are defined.
354             * @param multipartRoot The branch in the email structure in which we'll stuff the attachments.
355             * @throws MessagingException See {@link MimeMultipart#addBodyPart(BodyPart)} and
356             *             {@link #getBodyPartFromDatasource(AttachmentResource, String)}
357             */
358            private void setAttachments(final Email email, final MimeMultipart multipartRoot)
359                            throws MessagingException {
360                    for (final AttachmentResource resource : email.getAttachments()) {
361                            multipartRoot.addBodyPart(getBodyPartFromDatasource(resource, Part.ATTACHMENT));
362                    }
363            }

365            /**
366             * Helper method which generates a {@link BodyPart} from an {@link AttachmentResource} (from its {@link DataSource}) and a disposition
367             * type ({@link Part#INLINE} or {@link Part#ATTACHMENT}). With this the attachment data can be converted into objects that fit in the
368             * email structure. <br />
369             * <br />
370             * For every attachment and embedded image a header needs to be set.
371             * 
372             * @param resource An object that describes the attachment and contains the actual content data.
373             * @param dispositionType The type of attachment, {@link Part#INLINE} or {@link Part#ATTACHMENT} .
374             * @return An object with the attachment data read for placement in the email structure.
375             * @throws MessagingException All BodyPart setters.
376             */
377            private BodyPart getBodyPartFromDatasource(final AttachmentResource resource, final String dispositionType)
378                            throws MessagingException {
379                    final BodyPart attachmentPart = new MimeBodyPart();
380                    final DataSource ds = resource.getDataSource();
381                    // setting headers isn't working nicely using the javax mail API, so let's do that manually
382                    attachmentPart.setDataHandler(new DataHandler(resource.getDataSource()));
383                    attachmentPart.setFileName(resource.getName());
384                    attachmentPart.setHeader("Content-Type", ds.getContentType() + "; filename=" + ds.getName() + "; name=" + ds.getName());
385                    attachmentPart.setHeader("Content-ID", String.format("<%s>", ds.getName()));
386                    attachmentPart.setDisposition(dispositionType + "; size=0");
387                    return attachmentPart;
388            }

390            /**
391             * This class conveniently wraps all necessary mimemessage parts that need to be filled with content, attachments etc. The root is
392             * ultimately sent using JavaMail.<br />
393             * <br />
394             * The constructor creates a new email message constructed from {@link MimeMultipart} as follows:
395             * 
396             * <pre>
397             * - root
398             *      - related
399             *              - alternative
400             *                      - mail tekst
401             *                      - mail html tekst
402             *              - embedded images
403             *      - attachments
404             * </pre>
405             * 
406             * @author Benny Bottema
407             */
408            private class MimeEmailMessageWrapper {
409    
410                    private final MimeMultipart multipartRoot;
411    
412                    private final MimeMultipart multipartRelated;
413    
414                    private final MimeMultipart multipartAlternativeMessages;
415    
416                    /**
417                     * Creates an email skeleton structure, so that embedded images, attachments and (html) texts are being processed properly.
418                     */
419                    MimeEmailMessageWrapper() {
420                            multipartRoot = new MimeMultipart("mixed");
421                            final MimeBodyPart contentRelated = new MimeBodyPart();
422                            multipartRelated = new MimeMultipart("related");
423                            final MimeBodyPart contentAlternativeMessages = new MimeBodyPart();
424                            multipartAlternativeMessages = new MimeMultipart("alternative");
425                            try {
426                                    // construct mail structure
427                                    multipartRoot.addBodyPart(contentRelated);
428                                    contentRelated.setContent(multipartRelated);
429                                    multipartRelated.addBodyPart(contentAlternativeMessages);
430                                    contentAlternativeMessages.setContent(multipartAlternativeMessages);
431                            } catch (final MessagingException e) {
432                                    logger.error(e.getMessage(), e);
433                                    throw new RuntimeException(e.getMessage(), e);
434                            }
435                    }
436            }
